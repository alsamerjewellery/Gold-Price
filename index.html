async function fetchCSV() { const timestamp = new Date().getTime(); // وقت حالي لتمييز كل طلب const response = await fetch(data.csv?t=${timestamp}); const csvText = await response.text(); const lines = csvText.trim().split('\n'); const headers = parseCSVLine(lines[0]); // بناء رأس الجدول const thead = document.getElementById('tableHead'); const trHead = document.createElement('tr'); headers.forEach(h => { const th = document.createElement('th'); th.textContent = h; trHead.appendChild(th); }); thead.appendChild(trHead); const tbody = document.getElementById('tableBody'); tbody.innerHTML = ''; // تأكد من مسح البيانات السابقة for (let i = 1; i < lines.length; i++) { const row = parseCSVLine(lines[i]); const tr = document.createElement('tr'); row.forEach(cell => { const td = document.createElement('td'); if (!isNaN(cell) && cell.trim() !== '') { td.textContent = cell; td.classList.add('number-cell'); } else { td.textContent = cell; td.classList.add('text-cell'); } tr.appendChild(td); }); tbody.appendChild(tr); } }
